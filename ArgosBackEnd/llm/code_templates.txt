import { cookies } from "next/headers";
import { revalidatePath } from "next/cache";

    I<Feature>,
    Create<Feature>Data,
    Update<Feature>Data
} from "@/app/(protected)/<feature>/types/<feature>.types";

const EXPRESS_BASE_URL = process.env.EXPRESS_BASE_URL;

export async function create<Feature>(
    data: Create<Feature>Data
): Promise<{ success: boolean; id?: number; error?: string }> {
    try {
        if (!EXPRESS_BASE_URL) {
            throw new Error("EXPRESS_BASE_URL is not defined");
        }

        const cookieStore = await cookies();
        const session = cookieStore.get('session')?.value;

        if (!session) {
            throw new Error("No session cookie");
        }

        const res = await fetch(`${EXPRESS_BASE_URL}/<endpoint>`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Cookie': `session=${session}`,
            },
            body: JSON.stringify(data),
        });

        const json = await res.json();

        if (!res.ok) {
            throw new Error(json.motive || "Failed to create");
        }

        revalidatePath('/<feature>');
        return { success: true, id: json.id };
    } catch (err) {
        console.error("Create error:", err);
        return {
            success: false,
            error: err instanceof Error ? err.message : "Unknown error"
        };
    }
}

import { useState, useEffect, useTransition } from "react";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogDescription,
  DialogFooter,
} from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { create<Feature> } from "@/app/(protected)/<feature>/actions/<feature>.actions";

interface Create<Feature>ModalProps {
  open: boolean;
  onOpenChange: (open: boolean) => void;
}

export default function Create<Feature>Modal({
  open,
  onOpenChange,
}: Create<Feature>ModalProps) {
  const [isPending, startTransition] = useTransition();
  const [error, setError] = useState<string | null>(null);
  const [formData, setFormData] = useState({
    name: "",
    // ... other fields
  });

  // Reset state when modal closes
  useEffect(() => {
    if (!open) {
      setError(null);
      setFormData({
        name: "",
      });
    }
  }, [open]);

  const handleChange = (field: keyof typeof formData, value: string) => {
    setFormData((prev) => ({ ...prev, [field]: value }));
  };

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();

    setError(null);
    startTransition(async () => {
      const result = await create<Feature>({
        name: formData.name,
      });
      if (result.success) {
        onOpenChange(false);
      } else {
        setError(result.error || "Error al crear");
      }
    });
  };

  const handleOpenChange = (newOpen: boolean) => {
    if (!newOpen && isPending) return;
    onOpenChange(newOpen);
  };

      <DialogContent className="sm:max-w-[425px]">

"use client";

import { useState, useEffect, useTransition } from "react";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogDescription,
  DialogFooter,
} from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { update<Feature>, get<Feature>ById } from "@/app/(protected)/<feature>/actions/<feature>.actions";

interface Update<Feature>ModalProps {
  <feature>Id: number | null;
  open: boolean;
  onOpenChange: (open: boolean) => void;
}

export default function Update<Feature>Modal({
  <feature>Id,
  open,
  onOpenChange,
}: Update<Feature>ModalProps) {
  const [isPending, startTransition] = useTransition();
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [formData, setFormData] = useState({
    name: "",
  });

  // Reset state when modal closes
  useEffect(() => {
    if (!open) {
      setError(null);
      setFormData({
        name: "",
      });
    }
  }, [open]);

  // Fetch data when modal opens
  useEffect(() => {
    if (open && <feature>Id) {
      setIsLoading(true);
      setError(null);
      get<Feature>ById(<feature>Id)
        .then((result) => {
          if (result.success && result.data) {
            setFormData({
              name: result.data.name || "",
            });
          } else {
            setError(result.error || "Error al cargar");
          }
        })
        .catch(() => {
          setError("Error al cargar");
        })
        .finally(() => {
          setIsLoading(false);
        });
    }
  }, [open, <feature>Id]);

  const handleChange = (field: keyof typeof formData, value: string) => {
    setFormData((prev) => ({ ...prev, [field]: value }));
  };

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (!<feature>Id) return;

    setError(null);
    startTransition(async () => {
      const result = await update<Feature>(<feature>Id, formData);
      if (result.success) {
        onOpenChange(false);
      } else {
        setError(result.error || "Error al actualizar");
      }
    });
  };

  const handleOpenChange = (newOpen: boolean) => {
    if (!newOpen && isPending) return;
    onOpenChange(newOpen);
  };

  return (
    <Dialog open={open} onOpenChange={handleOpenChange}>
      <DialogContent className="sm:max-w-[425px]">
        <DialogHeader>
          <DialogTitle>Actualizar</DialogTitle>
          <DialogDescription>
            Modifica los datos y guarda los cambios.
          </DialogDescription>
        </DialogHeader>

        {isLoading ? (
          <div className="flex items-center justify-center py-8">
            <div className="h-8 w-8 animate-spin rounded-full border-4 border-primary border-t-transparent" />
          </div>
        ) : (
          <form onSubmit={handleSubmit}>
            <div className="grid gap-4 py-4">
              <div className="grid gap-2">
                <Label htmlFor="update-name">Nombre</Label>
                <Input
                  id="update-name"
                  value={formData.name}
                  onChange={(e) => handleChange("name", e.target.value)}
                  disabled={isPending}
                  required
                />
              </div>

              {error && (
                <p className="text-sm text-destructive">{error}</p>
              )}
            </div>

            <DialogFooter>
              <Button
                type="button"
                variant="outline"
                onClick={() => onOpenChange(false)}
                disabled={isPending}
              >
                Cancelar
              </Button>
              <Button type="submit" disabled={isPending}>
                {isPending ? "Guardando..." : "Guardar Cambios"}
              </Button>
            </DialogFooter>
          </form>
        )}
      </DialogContent>
    </Dialog>
  );
}

import MysqlClient from '../connections/mysqldb.js';

// CREATE
export async function create<Feature>(req, res) {
  try {
    const { name } = req.body || {};

    if (!name) {
      return res.status(400).json({ success: false, motive: 'name is required' });
    }

    const [result] = await MysqlClient.execute(
      'INSERT INTO <table> (name) VALUES (?)',
      [name]
    );

    return res.status(201).json({
      success: true,
      id: result.insertId,
      motive: 'Created successfully'
    });
  } catch (error) {
    console.error('Error creating:', error);
    return res.status(500).json({ success: false, motive: 'Server Error' });
  }
}

// READ - all
export async function get<Feature>s(req, res) {
  try {
    const [rows] = await MysqlClient.execute('SELECT * FROM <table> ORDER BY id DESC');
    return res.status(200).json({ success: true, data: rows });
  } catch (error) {
    console.error('Error getting:', error);
    return res.status(500).json({ success: false, motive: 'Server Error' });
  }
}

// READ - by ID
export async function get<Feature>ById(req, res) {
  try {
    const { id } = req.params;
    const [rows] = await MysqlClient.execute(
      'SELECT * FROM <table> WHERE id = ? LIMIT 1',
      [id]
    );

    if (rows.length === 0) {
      return res.status(404).json({ success: false, motive: 'Not found' });
    }

    return res.status(200).json({ success: true, data: rows[0] });
  } catch (error) {
    console.error('Error getting by ID:', error);
    return res.status(500).json({ success: false, motive: 'Server Error' });
  }
}

// UPDATE
export async function update<Feature>(req, res) {
  try {
    const { id } = req.params;
    const { name } = req.body || {};

    const [exists] = await MysqlClient.execute(
      'SELECT id FROM <table> WHERE id = ? LIMIT 1',
      [id]
    );
    if (exists.length === 0) {
      return res.status(404).json({ success: false, motive: 'Not found' });
    }

    const fields = [];
    const params = [];

    if (name !== undefined) {
      fields.push('name = ?');
      params.push(name);
    }

    if (fields.length === 0) {
      return res.status(400).json({ success: false, motive: 'No fields to update' });
    }

    params.push(id);

    const [result] = await MysqlClient.execute(
      `UPDATE <table> SET ${fields.join(', ')} WHERE id = ?`,
      params
    );

    if (result.affectedRows === 0) {
      return res.status(500).json({ success: false, motive: 'No record was updated' });
    }

    return res.status(200).json({ success: true, motive: 'Updated successfully' });
  } catch (error) {
    console.error('Error updating:', error);
    return res.status(500).json({ success: false, motive: 'Server Error' });
  }
}

// DELETE
export async function delete<Feature>(req, res) {
  try {
    const { id } = req.params;

    const [exists] = await MysqlClient.execute(
      'SELECT id FROM <table> WHERE id = ? LIMIT 1',
      [id]
    );
    if (exists.length === 0) {
      return res.status(404).json({ success: false, motive: 'Not found' });
    }

    const [result] = await MysqlClient.execute('DELETE FROM <table> WHERE id = ?', [id]);

    if (result.affectedRows === 0) {
      return res.status(500).json({ success: false, motive: 'No record was deleted' });
    }

    return res.status(200).json({ success: true, motive: 'Deleted successfully' });
  } catch (error) {
    console.error('Error deleting:', error);
    return res.status(500).json({ success: false, motive: 'Server Error' });
  }
}
